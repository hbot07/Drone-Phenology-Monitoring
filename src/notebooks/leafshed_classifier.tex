\section{Inputs and Data Model}

This document formalises the \emph{leaf-shed classifier} used to label tracked trees as \emph{deciduous} vs \emph{evergreen} from multi-date RGB orthomosaic crown chips.

Let $\mathcal{T}$ be the set of tracked trees (consensus crowns), and let $\mathcal{O}=\{o_1,\dots,o_T\}$ be the ordered set of orthomosaic indices (OM IDs).

For each tree $t\in\mathcal{T}$ and orthomosaic $o\in\mathcal{O}$, the pipeline extracts an RGB patch (\emph{crown chip})
\begin{equation}
P_{t,o} \in \mathbb{R}^{H\times W\times 3},
\end{equation}
by masking the orthomosaic with the consensus crown polygon and cropping to polygon bounds.

\paragraph{Pixel scaling.}
We assume pixels are converted to floating-point and scaled to $[0,1]$ per channel:
\begin{equation}
(r_p,g_p,b_p) \in [0,1]^3.
\end{equation}
(If the source is 8-bit RGB, this is typically done by dividing by 255. We intentionally omit any \texttt{clip()} / clamping details here.)

\section{Time-Series Signals}
\label{sec:signals}

The classifier uses three per-OM signals computed from each crown chip:
\begin{itemize}[leftmargin=*, itemsep=2pt]
  \item \texttt{veg\_fraction\_hsv} ($v_{t,o}$)
  \item \texttt{gcc\_mean} ($g_{t,o}$)
  \item \texttt{laplacian\_var} ($\ell_{t,o}$)
\end{itemize}

\subsection{HSV vegetation fraction: \texttt{veg\_fraction\_hsv}}

Convert each pixel $(r_p,g_p,b_p)$ to HSV $(h_p,s_p,v_p)\in[0,1]^3$.
Define a vegetation-like pixel indicator using fixed thresholds:
\begin{equation}
\mathbb{1}_{\mathrm{veg}}(p)=\mathbb{1}[0.18\le h_p\le 0.48] \;\mathbb{1}[s_p\ge 0.15] \;\mathbb{1}[v_p\ge 0.12].
\end{equation}
Then the observation-level vegetation fraction is
\begin{equation}
\texttt{veg\_fraction\_hsv}(t,o)=v_{t,o}=\frac{1}{N}\sum_{p=1}^{N}\mathbb{1}_{\mathrm{veg}}(p),\qquad N=HW.
\end{equation}

\subsection{Green chromatic coordinate mean: \texttt{gcc\_mean}}

Define $S_p=r_p+g_p+b_p$ and per-pixel GCC:
\begin{equation}
\mathrm{GCC}_p = \frac{g_p}{S_p+\varepsilon},\qquad \varepsilon>0.
\end{equation}
Then
\begin{equation}
\texttt{gcc\_mean}(t,o)=g_{t,o}=\frac{1}{N}\sum_{p=1}^{N} \mathrm{GCC}_p.
\end{equation}

\subsection{Texture proxy: \texttt{laplacian\_var}}

A sharpness/texture proxy is computed as the variance of the Laplacian of a grayscale image:
\begin{equation}
\texttt{laplacian\_var}(t,o)=\ell_{t,o} = \mathrm{Var}(\Delta\,\mathrm{gray}).
\end{equation}
This signal is used only through its \emph{amplitude over time}.

\section{From Discrete Observations to Per-Tree Time Series}
\label{sec:interpolation}

For each tree $t$, each signal produces a length-$T$ sequence across OMs:
\begin{equation}
\mathbf{v}_t = (v_{t,o_1},\dots,v_{t,o_T}),\quad
\mathbf{g}_t = (g_{t,o_1},\dots,g_{t,o_T}),
\quad
\mathbf{\ell}_t = (\ell_{t,o_1},\dots,\ell_{t,o_T}).
\end{equation}

Some observations may be missing or discarded by upstream quality control; denote missing entries as NaN.

\subsection{Why interpolate?}

The raw signals are observed only at the available OMs and can contain gaps (e.g., missing chips or QC exclusions). However, downstream steps require a coherent per-tree trajectory:
\begin{itemize}[leftmargin=*, itemsep=2pt]
  \item \textbf{Amplitude and minima/maxima:} DS uses per-tree ranges and minima of $\hat{v}_t$ and $\hat{g}_t$; these are unstable if intermediate OMs are missing.
  \item \textbf{Phenophase timeline:} we assign a state at each OM index, so we need values at each OM to apply the thresholds consistently.
\end{itemize}
Linear interpolation is therefore used as a simple imputation that preserves the seasonal trend without introducing extra model assumptions.

\subsection{Linear interpolation with edge fill}

For a generic per-tree series $\mathbf{x}_t$ observed at OM indices $\{o_1,\dots,o_T\}$:
\begin{itemize}[leftmargin=*, itemsep=2pt]
  \item If at least two entries are finite, linearly interpolate across OMs, and hold boundary values constant outside the observed range.
  \item If exactly one entry is finite, fill all OMs with that value.
  \item If no entries are finite, fill all OMs with 0.
\end{itemize}
We write the interpolated series as $\hat{\mathbf{x}}_t$.

\subsection{Per-tree min--max normalisation (shape-only series)}

For the vegetation signal only, define a shape-normalised series
\begin{equation}
\bar{v}_t[k] =
\begin{cases}
\dfrac{\hat{v}_t[k]-\min_j\hat{v}_t[j]}{\max_j\hat{v}_t[j]-\min_j\hat{v}_t[j]} & \text{if range}>10^{-9},\\
0 & \text{otherwise,}
\end{cases}
\end{equation}
where $k\in\{1,\dots,T\}$ is the OM time index.

\section{Deciduousness Score (DS)}
\label{sec:ds}

The DS is a convex-like combination of interpretable components, each clipped to $[0,1]$.

\subsection{Cross-tree amplitude normalisers (90th percentile)}

Define per-tree amplitudes on interpolated series:
\begin{align}
A_t^{\mathrm{veg}} &= \max_k \hat{v}_t[k] - \min_k \hat{v}_t[k],\\
A_t^{\mathrm{gcc}} &= \max_k \hat{g}_t[k] - \min_k \hat{g}_t[k],\\
A_t^{\mathrm{tex}} &= \max_k \hat{\ell}_t[k] - \min_k \hat{\ell}_t[k].
\end{align}
Let $A_{90}^{\mathrm{veg}}$ be the 90th percentile of $\{A_t^{\mathrm{veg}}\}_{t\in\mathcal{T}}$ (ignoring NaNs), and similarly $A_{90}^{\mathrm{gcc}}$ and $A_{90}^{\mathrm{tex}}$.

\subsection{Per-tree component scores}

Let $\tau_{\mathrm{veg}}=0.45$ be the vegetation minimum threshold.
Define
\begin{align}
 s_{\mathrm{veg\_amp}}(t) &= \min\Bigl(1,\, \frac{A_t^{\mathrm{veg}}}{A_{90}^{\mathrm{veg}}}\Bigr),\\
 s_{\mathrm{gcc\_amp}}(t) &= \min\Bigl(1,\, \frac{A_t^{\mathrm{gcc}}}{A_{90}^{\mathrm{gcc}}}\Bigr),\\
 s_{\mathrm{depth}}(t) &=
\begin{cases}
\dfrac{\tau_{\mathrm{veg}}-\min_k \hat{v}_t[k]}{\tau_{\mathrm{veg}}} & \text{if } \min_k \hat{v}_t[k] < \tau_{\mathrm{veg}},\\
0 & \text{otherwise,}
\end{cases}\\
 s_{\mathrm{tex}}(t) &= \min\Bigl(1,\, \frac{A_t^{\mathrm{tex}}}{A_{90}^{\mathrm{tex}}}\Bigr).
\end{align}

\subsection{Score definition and decision rule}

The implemented score is
\begin{equation}
\mathrm{DS}(t)=0.35\,s_{\mathrm{veg\_amp}}(t)+0.30\,s_{\mathrm{depth}}(t)+0.25\,s_{\mathrm{gcc\_amp}}(t)+0.10\,s_{\mathrm{tex}}(t).
\end{equation}

A tree is classified as deciduous iff
\begin{equation}
\mathrm{is\_deciduous}(t)=\mathbb{1}[\mathrm{DS}(t)\ge \tau_{\mathrm{DS}}],\qquad \tau_{\mathrm{DS}}=0.70.
\end{equation}

\section{Phenophase Labels and Event Timing}
\label{sec:phenophase}

Phenophase states are derived from the vegetation shape series $\bar{v}_t$ using thresholds
\begin{equation}
\tau_{\mathrm{on}}=0.65,\qquad \tau_{\mathrm{off}}=0.35.
\end{equation}
For each OM time index $k$:
\begin{equation}
\mathrm{state}_t[k]=
\begin{cases}
\texttt{stable} & \text{if } \mathrm{is\_deciduous}(t)=0,\\
\texttt{leaf\_on} & \text{if } \bar{v}_t[k]\ge \tau_{\mathrm{on}},\\
\texttt{leaf\_off} & \text{if } \bar{v}_t[k]\le \tau_{\mathrm{off}},\\
\texttt{transitioning} & \text{otherwise.}
\end{cases}
\end{equation}

The deciduous gallery is a direct visual check: it shows the same crown across OMs with a phenophase-coloured border.

\subsection{Key event OMs (deciduous trees only)}

Let the leaf-off trough index be
\begin{equation}
 k^{\mathrm{off}} = \arg\min_k \hat{v}_t[k].
\end{equation}
Then
\begin{equation}
\mathrm{full\_leaf\_off\_om}(t)=o_{k^{\mathrm{off}}}.
\end{equation}

Define \texttt{leaf\_off\_start\_om} as the OM \emph{right after} the last \texttt{leaf\_on} state before the trough, if such a state exists.
Define \texttt{leaf\_on\_return\_om} as the first \texttt{leaf\_on} OM at or after the trough, if it exists.
